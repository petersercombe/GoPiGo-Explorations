import pygame

def init():
    pygame.init()
    win = pygame.display.set_mode((300,150))
    # Place some text on the pygame screen:
    font = pygame.font.SysFont('sans', 30)
    textSurface = font.render('Keyboard Control', True, (255, 255, 255))
    win.blit(textSurface, (50, 50))
    pygame.display.update()

def keyPress(keyName):
    for event in pygame.event.get(): pass # Update events
    keyInput = pygame.key.get_pressed()
    myKey = getattr(pygame,'K_{}'.format(keyName))
    if keyInput[myKey]:
        return True
    else:
        return False

def main():
    if keyPress('w'):
        throttle = 1
    elif keyPress('s'):
        throttle = -1
    else: throttle = 0
    if keyPress('h'):
        steering=-1
    elif keyPress('j'):
        steering=-0.5
    elif keyPress('k'):
        steering=0.5
    elif keyPress('l'):
        steering=1
    else: steering=0
    return throttle, steering

# For testing only
if __name__ == '__main__':
    init()
    while True:
        print(main())
